drop table product;
drop table customer;
drop table online_sales;
drop table contract_sales;

create table customer
	(ID varchar(6),
    NAME varchar(20) not null,
    phone varchar(9),
    acoount_num varchar(9),
    card_num varchar(12),
    Address varchar(40)
    );
create table product
	(product_id varchar(8),
    type varchar (10),
    manufacturer varchar(10),
    stored_location varchar(8),
    price int,
    amount int
    );
create table online_sales
	(sales_num varchar(10),
    product_id varchar(8),
    ship_num varchar(4),
    ID varchar(6),
    price int,
    NAME varchar(20) not null,
    phone varchar(9),
    car_num varchar(12),
    Address varchar(40),
    amount int,
    arrivedate varchar(8),
    orderdate varchar(8),
    expectdate varchar(8)
    );
create table contact_sales
	(sales_num varchar(10),
    product_id varchar(8),
    ID varchar(6),
    name varchar(20) not null,
    phone varchar(9),
    account_num varchar(9),
    orderdate varchar(8),
    amount int
	);
